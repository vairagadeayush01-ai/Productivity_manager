           LearningEntry.created_at >= today.isoformat()  # BUG: String comparison!

✅ AFTER:
   @router.get("/today")
   async def get_today(db: Session = Depends(get_db)):
       today   = date_type.today()
       today_start = datetime.combine(today, time.min)  # NEW: Proper datetime
       entries = db.query(LearningEntry).filter(
           LearningEntry.created_at >= today_start  # FIXED: DateTime comparison
       ).order_by(LearningEntry.created_at.desc()).all()
       return {"date": today.isoformat(), "count": len(entries), "entries": [
           {"id": e.id, "source_type": e.source_type, "title": e.title,
            "summary": e.summary, "topics": (e.topics.split(", ") if e.topics else []),  # FIXED: Safe null
            "created_at": e.created_at.isoformat()}
           for e in entries
       ]}
